Цитата
HTOTAL = H_DISP + H_FPORCH + H_SYNC + H_BPORCH;
VTOTAL = V_DISP + V_FPORCH + V_SYNC + V_BPORCH;
if (INTERLACE == "1") VTOTAL = V_DISP + 2*V_FPORCH + 2*V_SYNC + 2*V_BPORCH;
HFREQ = (1000 * PIXEL_CLK) / HTOTAL;
VFREQ = (1000000 * PIXEL_CLK) / (HTOTAL * VTOTAL);
FIELDRATE = VFREQ;
if (INTERLACE == "1") VFREQ = 2*VFREQ;
HSYNCTIME = H_SYNC/PIXEL_CLK;
VSYNCTIME = (V_SYNC * HTOTAL) / PIXEL_CLK;
